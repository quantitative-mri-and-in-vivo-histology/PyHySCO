112	def eval(self, yc, yref=None, do_derivative=False, calc_hessian=False):
113		"""
114		Evaluates the objective function given the field map yc.
115
116		J(yc) = D(I1(yc),I2(yc)) + alpha S(yc) + beta P(yc) + rho Q(yc, yref)
117
118		If do_derivative is True, returns the gradient as well.
119
120		If calc_hessian is True, returns the Hessian and PC as well.
121
122		Parameters
123		----------
124		yc : torch.Tensor (size m_plus(m))
125			a field inhomogeneity map
126		yref : torch.Tensor (size m_plus(m)), optional
127			reference image used in proximal term, default is None
128		do_derivative : boolean, optional
129			flag to compute and return the gradient, default is False
130		calc_hessian : boolean, optional
131			flag to construct and return Hessian mat-vec, default is False
132
133		Returns
134		-------
135		J(yc) : torch.Tensor (size 1)
136			objective function value
137		dJ(yc) : torch.Tensor (size m_plus(m))
138			gradient of the objective function, only returned when do_derivative=True
139		H : Callable
140			callable matrix-vector product with (approximate) Hessian, only returned when calc_hessian=True
141		PC : Callable
142			callable solver to apply preconditioner, only returned when calc_hessian=True
143		"""
144		hd = torch.prod(self.dataObj.h)
145		
146		# compute interpolated image and derivative
147		T1c, Jac1, FI1, dFI1 = self.mp_transform(self.dataObj.I1, yc, do_derivative=do_derivative)
148		T2c, Jac2, FI2, dFI2 = self.mp_transform(self.dataObj.I2, -yc, do_derivative=do_derivative)
149
150		# compute distance measure
151		Dc, dDc = self.distance(T1c, T2c)
152		dD = None
153		if do_derivative:
154			geom = FI1 + FI2
155			intensity = Jac1 * dFI1 + Jac2 * dFI2
156			dD = self.D.transp_mat_mul(dDc*geom) + self.A.transp_mat_mul(dDc*intensity)
157		else:
158			geom = None
159			intensity = None
160
161		# smoothness regularizer
162		Sc, dS, d2S = self.S.eval(yc, do_derivative=do_derivative)
163
164		# intensity regularizer
165		G, dG, d2G = self.phi_EPI(Jac1-1, do_derivative=do_derivative, calc_hessian=calc_hessian)
166		Pc = torch.sum(G)
167		dP = None
168		if do_derivative:
169			dP = self.D.transp_mat_mul(dG)
170
171		# compute proximal term
172		if self.rho >0:
173			Qc,dQ,d2Q = self.Q.eval(yc, 1.0, yref, do_derivative=do_derivative)
174		else:
175			Qc = 0.0
176			dQ = 0.0
177			d2Q = None
178
179		# save terms of objective function and corrected images
180		self.Dc = Dc
181		self.Sc = Sc
182		self.Pc = Pc
183		self.Qc = Qc
184		self.corr1 = T1c
185		self.corr2 = T2c
186
187		Jc = Dc + hd*self.alpha*Sc + hd*self.beta*Pc + self.rho*Qc
188		if not do_derivative:
189			return Jc
190		dJ = dD + hd*self.alpha*dS + hd*self.beta*dP + self.rho*dQ
191		if not calc_hessian:
192			return Jc, dJ
193		else:
194			def H(x):
195				""" Matrix-vector product between Hessian and a tensor x of size m_plus(m). """
196				Dx = self.D.mat_mul(x)
197				dr = geom*Dx + intensity*self.A.mat_mul(x)
198				dr_d2psi = dr * hd
199				if self.beta == 0:  # d2P is zeros
200					d2D = self.D.transp_mat_mul(dr_d2psi*geom) + self.A.transp_mat_mul(dr_d2psi*intensity)				
201					return d2D + hd*self.alpha*d2S.mat_mul(x) + hd*self.rho*x
202				else:
203					d2D = self.D.transp_mat_mul(dr_d2psi*geom + hd*self.beta*d2G*Dx) + self.A.transp_mat_mul(dr_d2psi*intensity)
204					return d2D + hd*self.alpha*d2S.mat_mul(x) + hd*self.rho*x
205			if self.PC is not None:
206				diagD, diagP, diagS = self.PC.getM(geom, intensity,hd, d2G, self.D, self.A, self.S.H, self.alpha, self.beta)
207				self.PC.M += hd*self.rho
208				M = lambda x: self.PC.eval(x)
209			else:
210				M = None
211
212			return Jc, dJ, H, M

214	def mp_transform(self, I, b, do_derivative=False):
215		"""
216		Applies the distortion correction model.
217
218		TI(xc) = I(xc + bc) * (1 + dbc)
219
220		If do_derivative is True, computes gradient information as well.
221
222		Parameters
223		----------
224		I : ImageModels.ImageModel
225			interpolating image model
226		b : torch.Tensor (size m_plus(m))
227			a field inhomogeneity map
228		do_derivative : boolean, optional
229			flag to compute and return the gradient, default is False
230
231		Returns
232		----------
233		TI : torch.Tensor (size m)
234			result of applying the correction model to image I using the field map b
235		Jac : torch.Tensor (size m)
236			mass preserving factor 1 + dbc
237		FI : torch.Tensor (size m)
238			result of I interpolated on xc + bc
239		dFI : torch.Tensor (size m)
240			derivative of applying image model interpolation, None when do_derivative=False
241		"""
242		bc = self.A.mat_mul(b)  # averaging matrix & translation vector
243		dbc = self.D.mat_mul(b)  # derivative matrix & derivative of deformation vector
244		Jac = 1+dbc  # determinant of the transform xc+bc
245		xt = self.xc + bc
246
247		if do_derivative:
248			FI, dFI = I.eval(xt, do_derivative=True)  # interpolation on deformed grid
249			TI = FI * Jac  # mass preserving factor
250
251		else:
252			FI = I.eval(xt)  # interpolation on deformed grid
253			TI = FI * Jac  # mass preserving factor
254			dFI = None
255
256		return TI, Jac, FI, dFI

258	def phi_EPI(self, x, do_derivative=False, calc_hessian=False):
259		"""
260		Barrier function for the intensity regularization term, applied element-wise.
261
262		phi(x) = -x^4 / ((x^2 - 1))
263
264		phi satisfies these important conditions:
265
266			* phi(x) > 0, for all x
267
268			* phi(|x| -> 1) -> infinity
269
270			* phi(0) = 0
271
272			* phi is convex
273
274			* phi(x) = phi(-x)
275
276		Parameters
277		----------
278		x : torch.Tensor (size m)
279			partial derivative of the field map
280		do_derivative : boolean, optional
281			flag to compute the first derivative (default is False)
282		calc_hessian : boolean, optional
283			flag to compute the second derivative (default is False)
284
285		Returns
286		----------
287		G : torch.Tensor (size m)
288			function value
289		dG : torch.Tensor (size m)
290			first derivative of the function, None if do_derivative=False
291		d2G : torch.Tensor (size m)
292			second derivative of the function, None if calc_hessian=False
293		"""
294		dG, d2G = None, None
295		# penalize values outside of (-1,1)
296		x[torch.abs(x) >= 1] = float('inf')
297		x2 = x*x
298		G = torch.nan_to_num(-(x2*x2) / (x2-1))
299		if do_derivative:
300			dG = torch.nan_to_num(-2*(x*x2)*(x2-2) / (x2-1)**2)
301		if calc_hessian:
302			d2G = torch.nan_to_num(-2 * x2 * (x2 * x2 - 3 * x2 + 6) / (x2 - 1) ** 3)
303		return G, dG, d2G

321	def distance(self, Tc, Rc):
322		"""
323		Computes the sum of squared difference metric and derivatives for two images.
324
325		Parameters
326		----------
327		Tc : torch.Tensor (size m)
328			first image
329		Rc : torch.Tensor (size m)
330			second image
331
332		Returns
333		----------
334		Dc : torch.Tensor (size 1)
335			distance value = 1/2 * hd * rc.T * rc
336		dD : torch.Tensor (size m)
337			distance derivative = hd * dr
338		"""
339		d2psi = torch.prod(self.dataObj.h)
340		rc = Tc - Rc
341		Dc = 0.5 * d2psi * torch.norm(rc)**2
342		dr = 1
343		dD = rc*d2psi*dr
344		return Dc, dD

Defines and implements a distortion correction model and optimization problem for Echo-Planar MRI susceptibility artifact distortion correction.
